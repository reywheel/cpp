#include <iostream>

using namespace std;

int amountNumbersRecursion(int number);
int amountNumbers(int number);

int main() {
	setlocale(0, "");

	cout << "¬ведите число: " << endl;
	cin >> n;
	cout << amountNumbersRecursion(n) << endl;
	cout << amountNumbers(n) << endl;

int amountNumbersRecursion(int number) {
	int count = 0;
	if (number != 0) {
		count = amountNumbersRecursion(number / 10);
		count++;
	}
	return count;
}

int amountNumbers(int number) {
	int count = 0;
	while (number != 0) {
		count++;
		number /= 10;
	}
	return count;
}
