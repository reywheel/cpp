#include <iostream>
using namespace std;

bool flag;
int k, mass[100], i;




int main()
{
	setlocale(0, "");
    cout << "Введите размер одномерного массива: ";
	cin >> k;
	cout << "Введите массив по одному элементу: ";
	for (i = 0; i < k; i++) {
		cin >> mass[i];
	}

	do
	{
		flag = true;
		for (i = 1; i < k; i++) {
			if (mass[i] < mass[i - 1]) {
				mass[i - 1] = mass[i - 1] + mass[i];
				mass[i] = mass[i - 1] - mass[i];
				mass[i - 1] = mass[i - 1] - mass[i];
				flag = false;
			}
		}

	} while (!flag);

	cout << "Вывод массива: ";
	for (i = 0; i < k; i++) {
		cout << mass[i] << " ";
	}

	return(1);
}
